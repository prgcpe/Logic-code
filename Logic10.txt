#include <stdio.h>
#include <string.h>
void separate01(char[], int);
int main(){
	char s[] = "1001001";
	int length = strlen(s);
	
	separate01(s, length);
	printf("%s\n", s);
	return 0;
}
void separate01(char s[], int len){
	int i, count0, count1;
	count0 = count1 = 0;
	
	for(i = 0; i < len; i++){
		if(s[i] == '0')
			count0++;
		else
			count1++;
	}
	for(i = 0; i < len; i++)
		s[i] = '0';
	for(i = count0; i < len; i++)
		s[i] = '1';
	
}

	int i, j = 0;
	
	for(i = 0; i < len; i++) {
		if(s[i] == '0')
			s[j++] = '0';
	}
	for(i = j; i < len; i++)

int isIsogram(char str[], int len){
	int count[26] = {0}, i;
	for(i = 0; i < len; i++)
		count[str[i] - 'a']++;
	
	for(i = 0; i < 26; i++){
		if(count[i] > 1)
			return 0;
	}
	return 1;
}
