void countDigit(int n){
	int count = 0;
	while(n > 0){
		count++;
		n /= 10;
	}
	if(count > 1)
		printf("%d-digits ", count);
	else
		printf("1-digit");
}
void countEvenOdd(int n){
	int counteven = 0, countodd = 0;
	while(n > 0){
		if((n % 10) % 2 == 0)
			counteven++;
		else
			countodd++;
		n /= 10;
	}
	if(counteven == 0)
		printf("all odd ");
	else if(countodd == 0)
		printf("all even ");
	else
		printf("mix ");
}
void isPalindrome(int n){
	if(n == reverse(n))
		printf("palindrome");
	else
		printf("number");
}
int reverse(int n){
	int rev = 0;
	while(n > 0){
		rev = rev * 10 + n % 10;
		n /= 10;
	}
	return rev;
}

#include <stdio.h>
int toDecimal(int);
int isBinary(int);
int main(){
	int num, decimal;
	printf("Enter a number : ");
	scanf("%d", &num);
	if(isBinary(num)){
		decimal = toDecimal(num);
		printf("binary %d is %d in decimal", num, decimal);
	}
	else
		printf("not binary");
	return 0;
}
int isBinary(int n){
	
	while(n > 0){
		if(n % 10 > 1)
			return 0;
		n /= 10;
	}
	return 1;
}
int toDecimal(int n){
	int base = 1, decimal = 0;
	while(n > 0){
		decimal = decimal + base * (n % 10);
		base *= 2;
		n /= 10;
	}
	return decimal;
}

		printf("prime\n");

		printf("palindrome\n");
	}

		ordinary(num);

int isPrime(int n){
	int i;
	if(n < 2)
		return 0;
	if(n == 2 || n == 3)
		return 1;
	if(n % 2 == 0 || n % 3 == 0)
		return 0;
	for(i = 5; i*i <= n; i+=6){
		if(n % i == 0 || n % (i + 2) == 0)
			return 0;
	}
	return 1;
}
int isPalindrome(int n){
	if(reverse(n) == n)
		return 1;
	else
		return 0;
}
int reverse(int n){
	int rev = 0;
	while(n > 0){
		rev = rev * 10 + n % 10;
		n /= 10; 
	}
	return rev;
}
void ordinary(int n){
	if(n % 2 == 0)
		printf("ordinary even number\n");
	else
		printf("ordinary odd number\n");

Enter numerator : 3
Enter demominator : 8 
3/8 = 3/8

#include <stdio.h>
int gcd(int, int);
int main(){
	int num, demo, g;
	printf("Enter numerator : ");
	scanf("%d", &num);
	printf("Enter demominator : ");
	scanf("%d", &demo);
	g = gcd(num, demo);
	if(num % demo == 0)
		printf("%d/%d = %d", num, demo, num / demo);
	else
		printf("%d/%d = %d/%d", num, demo, num/g, demo/g);
	return 0;
}
int gcd(int a, int b){
	int temp;
	while(a % b != 0){
		temp = a;
		a = b;
		b = temp % b;
	}
	return b;
}
